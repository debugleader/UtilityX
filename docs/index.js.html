const createVars = (name, amount, value = 'unspecified') => {
	let variables = [];
	for (let i = 0; i !== amount; i++) {
		if (typeof value !== 'object' || value == null) {
			eval(`${name}${i} = '${value}'`);
			variables.push(name + i);
		} else {
			eval(`${name}${i} = '${value[i]}'`);
			variables.push(name + i);
		}
	}
	return variables;
};
